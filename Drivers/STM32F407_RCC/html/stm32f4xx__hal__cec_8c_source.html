389 HAL_StatusTypeDef HAL_CEC_Transmit_IT(CEC_HandleTypeDef *hcec, uint8_t InitiatorAddress, uint8_t DestinationAddress, uint8_t *pData, uint32_t Size)
390 {
391  /* if the IP isn't already busy and if there is no previous transmission
392  already pending due to arbitration lost */
393  if (hcec->gState == HAL_CEC_STATE_READY)
394  {
395  if((pData == NULL ) && (Size > 0U))
396  {
397  return HAL_ERROR;
398  }
399 
400  assert_param(IS_CEC_ADDRESS(DestinationAddress));
401  assert_param(IS_CEC_ADDRESS(InitiatorAddress));
402  assert_param(IS_CEC_MSGSIZE(Size));
403 
404  /* Process Locked */
405  __HAL_LOCK(hcec);
406  hcec->pTxBuffPtr = pData;
407  hcec->gState = HAL_CEC_STATE_BUSY_TX;
408  hcec->ErrorCode = HAL_CEC_ERROR_NONE;
409 
410  /* initialize the number of bytes to send,
411  * 0 means only one header is sent (ping operation) */
412  hcec->TxXferCount = Size;
413 
414  /* in case of no payload (Size = 0), sender is only pinging the system;
415  Set TX End of Message (TXEOM) bit, must be set before writing data to TXDR */
416  if (Size == 0U)
417  {
418  __HAL_CEC_LAST_BYTE_TX_SET(hcec);
419  }
420 
421  /* send header block */
422  hcec->Instance->TXDR = ((uint8_t)(InitiatorAddress << CEC_INITIATOR_LSB_POS) |(uint8_t) DestinationAddress);
423  /* Set TX Start of Message (TXSOM) bit */
424  __HAL_CEC_FIRST_BYTE_TX_SET(hcec);
425 
426  /* Process Unlocked */
427  __HAL_UNLOCK(hcec);
428 
429  return HAL_OK;
430 
431  }

434  return HAL_BUSY;

464 void HAL_CEC_IRQHandler(CEC_HandleTypeDef *hcec)
465 {
466 
467  /* save interrupts register for further error or interrupts handling purposes */
468  uint32_t reg = 0U;
469  reg = hcec->Instance->ISR;
470 
471 
472  /* ----------------------------Arbitration Lost Management----------------------------------*/
473  /* CEC TX arbitration error interrupt occurred --------------------------------------*/
474  if((reg & CEC_FLAG_ARBLST) != RESET)
475  {
476  hcec->ErrorCode = HAL_CEC_ERROR_ARBLST;
477  __HAL_CEC_CLEAR_FLAG(hcec, CEC_FLAG_ARBLST);
478  }
479 
480  /* ----------------------------Rx Management----------------------------------*/
481  /* CEC RX byte received interrupt ---------------------------------------------------*/
482  if((reg & CEC_FLAG_RXBR) != RESET)
483  {
484  /* reception is starting */
485  hcec->RxState = HAL_CEC_STATE_BUSY_RX;
486  hcec->RxXferSize++;
487  /* read received byte */
488  *hcec->Init.RxBuffer++ = hcec->Instance->RXDR;
489  __HAL_CEC_CLEAR_FLAG(hcec, CEC_FLAG_RXBR);
490  }
491 
492  /* CEC RX end received interrupt ---------------------------------------------------*/
493  if((reg & CEC_FLAG_RXEND) != RESET)
494  {
495  /* clear IT */
496  __HAL_CEC_CLEAR_FLAG(hcec, CEC_FLAG_RXEND);
497 
498  /* Rx process is completed, restore hcec->RxState to Ready */
499  hcec->RxState = HAL_CEC_STATE_READY;
500  hcec->ErrorCode = HAL_CEC_ERROR_NONE;
501  hcec->Init.RxBuffer -= hcec->RxXferSize;
502  HAL_CEC_RxCpltCallback(hcec, hcec->RxXferSize);
503  hcec->RxXferSize = 0U;
504  }
505 
506  /* ----------------------------Tx Management----------------------------------*/
507  /* CEC TX byte request interrupt ------------------------------------------------*/
508  if((reg & CEC_FLAG_TXBR) != RESET)
509  {
510  if (hcec->TxXferCount == 0U)
511  {
512  /* if this is the last byte transmission, set TX End of Message (TXEOM) bit */
513  __HAL_CEC_LAST_BYTE_TX_SET(hcec);
514  hcec->Instance->TXDR = *hcec->pTxBuffPtr++;
515  }
516  else
517  {
518  hcec->Instance->TXDR = *hcec->pTxBuffPtr++;
519  hcec->TxXferCount--;
520  }
521  /* clear Tx-Byte request flag */
522  __HAL_CEC_CLEAR_FLAG(hcec,CEC_FLAG_TXBR);
523  }
524 
525  /* CEC TX end interrupt ------------------------------------------------*/
526  if((reg & CEC_FLAG_TXEND) != RESET)
527  {
528  __HAL_CEC_CLEAR_FLAG(hcec, CEC_FLAG_TXEND);
529 
530  /* Tx process is ended, restore hcec->gState to Ready */
531  hcec->gState = HAL_CEC_STATE_READY;
532  /* Call the Process Unlocked before calling the Tx call back API to give the possibility to
533  start again the Transmission under the Tx call back API */
534  __HAL_UNLOCK(hcec);
535  hcec->ErrorCode = HAL_CEC_ERROR_NONE;
536  HAL_CEC_TxCpltCallback(hcec);
537  }
538 
539  /* ----------------------------Rx/Tx Error Management----------------------------------*/
540  if ((reg & (CEC_ISR_RXOVR|CEC_ISR_BRE|CEC_ISR_SBPE|CEC_ISR_LBPE|CEC_ISR_RXACKE|CEC_ISR_TXUDR|CEC_ISR_TXERR|CEC_ISR_TXACKE)) != 0U)
541  {
542  hcec->ErrorCode = reg;
543  __HAL_CEC_CLEAR_FLAG(hcec, HAL_CEC_ERROR_RXOVR|HAL_CEC_ERROR_BRE|CEC_FLAG_LBPE|CEC_FLAG_SBPE|HAL_CEC_ERROR_RXACKE|HAL_CEC_ERROR_TXUDR|HAL_CEC_ERROR_TXERR|HAL_CEC_ERROR_TXACKE);
544 
545 
546  if((reg & (CEC_ISR_RXOVR|CEC_ISR_BRE|CEC_ISR_SBPE|CEC_ISR_LBPE|CEC_ISR_RXACKE)) != RESET)
547  {
548  hcec->Init.RxBuffer-=hcec->RxXferSize;
549  hcec->RxXferSize = 0U;
550  hcec->RxState = HAL_CEC_STATE_READY;
551  }
552  else if (((reg & (CEC_ISR_TXUDR|CEC_ISR_TXERR|CEC_ISR_TXACKE)) != RESET) && ((reg & CEC_ISR_ARBLST) == RESET))
553  {
554  /* Set the CEC state ready to be able to start again the process */
555  hcec->gState = HAL_CEC_STATE_READY;
556  }
557 
558  /* Error Call Back */
559  HAL_CEC_ErrorCallback(hcec);
560  }
