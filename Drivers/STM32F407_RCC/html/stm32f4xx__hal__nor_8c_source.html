958 HAL_NOR_StatusTypeDef HAL_NOR_GetStatus(NOR_HandleTypeDef *hnor, uint32_t Address, uint32_t Timeout)
959 {

961  uint16_t tmpSR1 = 0U, tmpSR2 = 0U;
962  uint32_t tickstart = 0U;
963 
964  /* Poll on NOR memory Ready/Busy signal ------------------------------------*/
965  HAL_NOR_MspWait(hnor, Timeout);
966 
967  /* Get the NOR memory operation status -------------------------------------*/
968 
969  /* Get tick */
970  tickstart = HAL_GetTick();
971  while((status != HAL_NOR_STATUS_SUCCESS ) && (status != HAL_NOR_STATUS_TIMEOUT))
972  {
973  /* Check for the Timeout */
974  if(Timeout != HAL_MAX_DELAY)
975  {
976  if((Timeout == 0U)||((HAL_GetTick() - tickstart ) > Timeout))
977  {
978  status = HAL_NOR_STATUS_TIMEOUT;
979  }
980  }
981 
982  /* Read NOR status register (DQ6 and DQ5) */
983  tmpSR1 = *(__IO uint16_t *)Address;
984  tmpSR2 = *(__IO uint16_t *)Address;
985 
986  /* If DQ6 did not toggle between the two reads then return HAL_NOR_STATUS_SUCCESS */
987  if((tmpSR1 & NOR_MASK_STATUS_DQ6) == (tmpSR2 & NOR_MASK_STATUS_DQ6))
988  {
989  return HAL_NOR_STATUS_SUCCESS ;
990  }
991 
992  if((tmpSR1 & NOR_MASK_STATUS_DQ5) == NOR_MASK_STATUS_DQ5)
993  {
994  status = HAL_NOR_STATUS_ONGOING;
995  }
996 
997  tmpSR1 = *(__IO uint16_t *)Address;
998  tmpSR2 = *(__IO uint16_t *)Address;
999 
1000  /* If DQ6 did not toggle between the two reads then return HAL_NOR_STATUS_SUCCESS */
1001  if((tmpSR1 & NOR_MASK_STATUS_DQ6) == (tmpSR2 & NOR_MASK_STATUS_DQ6))
1002  {
1003  return HAL_NOR_STATUS_SUCCESS;
1004  }

1006  {

1008  }
1009  }
