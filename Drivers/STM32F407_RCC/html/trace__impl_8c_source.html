110 // ITM is the ARM standard mechanism, running over SWD/SWO on Cortex-M3/M4
111 // devices, and is the recommended setting, if available.
112 //
113 // The JLink probe and the GDB server fully support SWD/SWO
114 // and the JLink Debugging plug-in enables it by default.
115 // The current OpenOCD does not include support to parse the SWO stream,
116 // so this configuration will not work on OpenOCD (will not crash, but
117 // nothing will be displayed in the output console).
118 
119 #if !defined(OS_INTEGER_TRACE_ITM_STIMULUS_PORT)
120 #define OS_INTEGER_TRACE_ITM_STIMULUS_PORT (0)
121 #endif
122 
123 static ssize_t
124 _trace_write_itm (const char* buf, size_t nbyte)
125 {
126  for (size_t i = 0; i < nbyte; i++)
127  {
128  // Check if ITM or the stimulus port are not enabled
129  if (((ITM->TCR & ITM_TCR_ITMENA_Msk) == 0)
130  || ((ITM->TER & (1UL << OS_INTEGER_TRACE_ITM_STIMULUS_PORT)) == 0))
131  {
132  return (ssize_t)i; // return the number of sent characters (may be 0)
133  }
134 
135  // Wait until STIMx is ready...
136  while (ITM->PORT[OS_INTEGER_TRACE_ITM_STIMULUS_PORT].u32 == 0)
137  ;
138  // then send data, one byte at a time
139  ITM->PORT[OS_INTEGER_TRACE_ITM_STIMULUS_PORT].u8 = (uint8_t) (*buf++);
140  }
141 
142  return (ssize_t)nbyte; // all characters successfully sent
143 }

155 // Semihosting is the other output channel that can be used for the trace
156 // messages. It comes in two flavours: STDOUT and DEBUG. The STDOUT channel
157 // is the equivalent of the stdout in POSIX and in most cases it is forwarded
158 // to the GDB server stdout stream. The debug channel is a separate
159 // channel. STDOUT is buffered, so nothing is displayed until a \n;
160 // DEBUG is not buffered, but can be slow.
161 //
162 // Choosing between semihosting stdout and debug depends on the capabilities
163 // of your GDB server, and also on specific needs. It is recommended to test
164 // DEBUG first, and if too slow, try STDOUT.
165 //
166 // The JLink GDB server fully support semihosting, and both configurations
167 // are available; to activate it, use "monitor semihosting enable" or check
168 // the corresponding button in the JLink Debugging plug-in.
169 // In OpenOCD, support for semihosting can be enabled using
170 // "monitor arm semihosting enable".
171 //
172 // Note: Applications built with semihosting output active normally cannot
173 // be executed without the debugger connected and active, since they use
174 // BKPT to communicate with the host. However, with a carefully written
175 // HardFault_Handler, the semihosting BKPT calls can be processed, making
176 // possible to run semihosting applications as standalone, without being
177 // terminated with hardware faults.
178 
179 #endif // OS_USE_TRACE_SEMIHOSTING_DEBUG_*

185 static ssize_t
186 _trace_write_semihosting_stdout (const char* buf, size_t nbyte)
187 {
188  static int handle;
189  void* block[3];
190  int ret;
191 
192  if (handle == 0)
193  {
194  // On the first call get the file handle from the host
195  block[0] = ":tt"; // special filename to be used for stdin/out/err
196  block[1] = (void*) 4; // mode "w"
197  // length of ":tt", except null terminator
198  block[2] = (void*) (sizeof(":tt") - 1);
199 
200  ret = call_host (SEMIHOSTING_SYS_OPEN, (void*) block);
201  if (ret == -1)
202  return -1;
203 
204  handle = ret;
205  }
206 
207  block[0] = (void*) handle;
208  block[1] = (void*) buf;
209  block[2] = (void*) nbyte;
210  // send character array to host file/device
211  ret = call_host (SEMIHOSTING_SYS_WRITE, (void*) block);
212  // this call returns the number of bytes NOT written (0 if all ok)
213 
214  // -1 is not a legal value, but SEGGER seems to return it
215  if (ret == -1)
216  return -1;
217 
218  // The compliant way of returning errors
219  if (ret == (int) nbyte)
220  return -1;
221 
222  // Return the number of bytes written
223  return (ssize_t) (nbyte) - (ssize_t) ret;
224 }
225 
226 #endif // OS_USE_TRACE_SEMIHOSTING_STDOUT

232 #define OS_INTEGER_TRACE_TMP_ARRAY_SIZE (16)
233 
234 static ssize_t
235 _trace_write_semihosting_debug (const char* buf, size_t nbyte)
236 {
237  // Since the single character debug channel is quite slow, try to
238  // optimise and send a null terminated string, if possible.
239  if (buf[nbyte] == '\0')
240  {
241  // send string
242  call_host (SEMIHOSTING_SYS_WRITE0, (void*) buf);
243  }
244  else
245  {
246  // If not, use a local buffer to speed things up
247  char tmp[OS_INTEGER_TRACE_TMP_ARRAY_SIZE];
248  size_t togo = nbyte;
249  while (togo > 0)
250  {
251  unsigned int n = ((togo < sizeof(tmp)) ? togo : sizeof(tmp));
252  unsigned int i = 0;
253  for (; i < n; ++i, ++buf)
254  {
255  tmp[i] = *buf;
256  }
257  tmp[i] = '\0';
258 
259  call_host (SEMIHOSTING_SYS_WRITE0, (void*) tmp);
260 
261  togo -= n;
262  }
263  }
264 
265  // All bytes written
266  return (ssize_t) nbyte;
267 }
