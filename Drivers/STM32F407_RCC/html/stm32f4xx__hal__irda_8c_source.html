897 void HAL_IRDA_IRQHandler(IRDA_HandleTypeDef *hirda)
898 {
899  uint32_t isrflags = READ_REG(hirda->Instance->SR);
900  uint32_t cr1its = READ_REG(hirda->Instance->CR1);
901  uint32_t cr3its = READ_REG(hirda->Instance->CR3);
902  uint32_t errorflags = 0x00U;
903  uint32_t dmarequest = 0x00U;
904 
905  /* If no error occurs */
906  errorflags = (isrflags & (uint32_t)(USART_SR_PE | USART_SR_FE | USART_SR_ORE | USART_SR_NE));
907  if(errorflags == RESET)
908  {
909  /* IRDA in mode Receiver -----------------------------------------------*/
910  if(((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
911  {
912  IRDA_Receive_IT(hirda);
913  return;
914  }
915  }
916 
917  /* If some errors occur */
918  if((errorflags != RESET) && ((cr3its & (USART_CR3_EIE | USART_CR1_PEIE)) != RESET))
919  {
920  /* IRDA parity error interrupt occurred -------------------------------*/
921  if(((isrflags & USART_SR_PE) != RESET) && ((cr1its & USART_CR1_PEIE) != RESET))
922  {
923  hirda->ErrorCode |= HAL_IRDA_ERROR_PE;
924  }
925 
926  /* IRDA noise error interrupt occurred --------------------------------*/
927  if(((isrflags & USART_SR_NE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
928  {
929  hirda->ErrorCode |= HAL_IRDA_ERROR_NE;
930  }
931 
932  /* IRDA frame error interrupt occurred --------------------------------*/
933  if(((isrflags & USART_SR_FE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
934  {
935  hirda->ErrorCode |= HAL_IRDA_ERROR_FE;
936  }
937 
938  /* IRDA Over-Run interrupt occurred -----------------------------------*/
939  if(((isrflags & USART_SR_ORE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
940  {
941  hirda->ErrorCode |= HAL_IRDA_ERROR_ORE;
942  }
943  /* Call IRDA Error Call back function if need be -----------------------*/
944  if(hirda->ErrorCode != HAL_IRDA_ERROR_NONE)
945  {
946  /* IRDA in mode Receiver ---------------------------------------------*/
947  if(((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
948  {
949  IRDA_Receive_IT(hirda);
950  }
951 
952  dmarequest = HAL_IS_BIT_SET(hirda->Instance->CR3, USART_CR3_DMAR);
953  /* If Overrun error occurs, or if any error occurs in DMA mode reception,
954  consider error as blocking */
955  if(((hirda->ErrorCode & HAL_IRDA_ERROR_ORE) != RESET) || dmarequest)
956  {
957  /* Blocking error : transfer is aborted
958  Set the IRDA state ready to be able to start again the process,
959  Disable Rx Interrupts, and disable Rx DMA request, if ongoing */
960  IRDA_EndRxTransfer(hirda);
961 
962  /* Disable the IRDA DMA Rx request if enabled */
963  if(HAL_IS_BIT_SET(hirda->Instance->CR3, USART_CR3_DMAR))
964  {
965  CLEAR_BIT(hirda->Instance->CR3, USART_CR3_DMAR);
966
