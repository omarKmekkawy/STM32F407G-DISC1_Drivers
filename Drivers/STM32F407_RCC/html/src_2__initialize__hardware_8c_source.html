65 // This is the application hardware initialisation routine,
66 // redefined to add more inits.
67 //
68 // Called early from _start(), right after data & bss init, before
69 // constructors.
70 //
71 // After Reset the Cortex-M processor is in Thread mode,
72 // priority is Privileged, and the Stack is set to Main.
73 //
74 // Warning: The HAL requires the system timer, running at 1000 Hz
75 // and calling HAL_IncTick().

113 void
114 __attribute__((weak))
115 SystemClock_Config(void)
116 {
117  // Enable Power Control clock
118  __PWR_CLK_ENABLE();
119 
120  // The voltage scaling allows optimizing the power consumption when the
121  // device is clocked below the maximum system frequency, to update the
122  // voltage scaling value regarding system frequency refer to product
123  // datasheet.
124  __HAL_PWR_VOLTAGESCALING_CONFIG(PWR_REGULATOR_VOLTAGE_SCALE1);
125 
126 #warning "Please check if the SystemClock_Config() settings match your board!"
127  // Comment out the warning after checking and updating.
128 
129  RCC_OscInitTypeDef RCC_OscInitStruct;
130 
131 #if defined(HSE_VALUE) && (HSE_VALUE != 0)
132  // Enable HSE Oscillator and activate PLL with HSE as source.
133  // This is tuned for STM32F4-DISCOVERY; update it for your board.
134  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSE;
135  RCC_OscInitStruct.HSEState = RCC_HSE_ON;
136  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSE;
137  // This assumes the HSE_VALUE is a multiple of 1 MHz. If this is not
138  // your case, you have to recompute these PLL constants.
139  RCC_OscInitStruct.PLL.PLLM = (HSE_VALUE/1000000u);
140 #else
141  // Use HSI and activate PLL with HSI as source.
142  // This is tuned for NUCLEO-F411; update it for your board.
143  RCC_OscInitStruct.OscillatorType = RCC_OSCILLATORTYPE_HSI;
144  RCC_OscInitStruct.HSIState = RCC_HSI_ON;
145  // 16 is the average calibration value, adjust for your own board.
146  RCC_OscInitStruct.HSICalibrationValue = 16;
147  RCC_OscInitStruct.PLL.PLLSource = RCC_PLLSOURCE_HSI;
148  // This assumes the HSI_VALUE is a multiple of 1 MHz. If this is not
149  // your case, you have to recompute these PLL constants.
150  RCC_OscInitStruct.PLL.PLLM = (HSI_VALUE/1000000u);
151 #endif
152 
153  RCC_OscInitStruct.PLL.PLLN = 336;
154 #if defined(STM32F401xC) || defined(STM32F401xE) || defined(STM32F411xE)
155  RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV4; /* 84 MHz */
156 #elif defined(STM32F427xx) || defined(STM32F437xx) || defined(STM32F429xx) || defined(STM32F439xx)
157  RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV2; /* 168 MHz */
158 #elif defined(STM32F405xx) || defined(STM32F415xx) || defined(STM32F407xx) || defined(STM32F417xx)
159  RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV2; /* 168 MHz */
160 #elif defined(STM32F446xx)
161  RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV2; /* 168 MHz */
162 #else
163  RCC_OscInitStruct.PLL.PLLP = RCC_PLLP_DIV4; /* 84 MHz, conservative */
164 #endif
165  RCC_OscInitStruct.PLL.PLLQ = 7; /* To make USB work. */
166  RCC_OscInitStruct.PLL.PLLState = RCC_PLL_ON;
167  HAL_RCC_OscConfig(&RCC_OscInitStruct);
168 
169  RCC_ClkInitTypeDef RCC_ClkInitStruct;
170  // Select PLL as system clock source and configure the HCLK, PCLK1 and PCLK2
171  // clocks dividers
172  RCC_ClkInitStruct.ClockType = (RCC_CLOCKTYPE_SYSCLK | RCC_CLOCKTYPE_HCLK
173  | RCC_CLOCKTYPE_PCLK1 | RCC_CLOCKTYPE_PCLK2);
174  RCC_ClkInitStruct.SYSCLKSource = RCC_SYSCLKSOURCE_PLLCLK;
175 #if defined(STM32F401xC) || defined(STM32F401xE) || defined(STM32F411xE)
176  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
177  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV2;
178  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV1;
179  HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_2);
180 #else
181  // This is expected to work for most large cores.
182  // Check and update it for your own configuration.
183  RCC_ClkInitStruct.AHBCLKDivider = RCC_SYSCLK_DIV1;
184  RCC_ClkInitStruct.APB1CLKDivider = RCC_HCLK_DIV4;
185  RCC_ClkInitStruct.APB2CLKDivider = RCC_HCLK_DIV2;
186  HAL_RCC_ClockConfig(&RCC_ClkInitStruct, FLASH_LATENCY_5);
187 #endif
188 
189  HAL_SYSTICK_Config(HAL_RCC_GetHCLKFreq()/1000);
190 
191  HAL_SYSTICK_CLKSourceConfig(SYSTICK_CLKSOURCE_HCLK);
192 }
