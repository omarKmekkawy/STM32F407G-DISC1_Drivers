769 void HAL_SMARTCARD_IRQHandler(SMARTCARD_HandleTypeDef *hsc)
770 {
771  uint32_t isrflags = READ_REG(hsc->Instance->SR);
772  uint32_t cr1its = READ_REG(hsc->Instance->CR1);
773  uint32_t cr3its = READ_REG(hsc->Instance->CR3);
774  uint32_t dmarequest = 0x00U;
775  uint32_t errorflags = 0x00U;
776 
777  /* If no error occurs */
778  errorflags = (isrflags & (uint32_t)(USART_SR_PE | USART_SR_FE | USART_SR_ORE | USART_SR_NE));
779  if(errorflags == RESET)
780  {
781  /* SMARTCARD in mode Receiver -------------------------------------------------*/
782  if(((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
783  {
784  SMARTCARD_Receive_IT(hsc);
785  return;
786  }
787  }
788 
789  /* If some errors occur */
790  if((errorflags != RESET) && ((cr3its & (USART_CR3_EIE | USART_CR1_PEIE)) != RESET))
791  {
792  /* SMARTCARD parity error interrupt occurred ---------------------------*/
793  if(((isrflags & SMARTCARD_FLAG_PE) != RESET) && ((cr1its & USART_CR1_PEIE) != RESET))
794  {
795  hsc->ErrorCode |= HAL_SMARTCARD_ERROR_PE;
796  }
797 
798  /* SMARTCARD frame error interrupt occurred ----------------------------*/
799  if(((isrflags & SMARTCARD_FLAG_FE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
800  {
801  hsc->ErrorCode |= HAL_SMARTCARD_ERROR_FE;
802  }
803 
804  /* SMARTCARD noise error interrupt occurred ----------------------------*/
805  if(((isrflags & SMARTCARD_FLAG_NE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
806  {
807  hsc->ErrorCode |= HAL_SMARTCARD_ERROR_NE;
808  }
809 
810  /* SMARTCARD Over-Run interrupt occurred -------------------------------*/
811  if(((isrflags & SMARTCARD_FLAG_ORE) != RESET) && ((cr3its & USART_CR3_EIE) != RESET))
812  {
813  hsc->ErrorCode |= HAL_SMARTCARD_ERROR_ORE;
814  }
815  /* Call the Error call Back in case of Errors */
816  if(hsc->ErrorCode != HAL_SMARTCARD_ERROR_NONE)
817  {
818  /* SMARTCARD in mode Receiver -----------------------------------------------*/
819  if(((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
820  {
821  SMARTCARD_Receive_IT(hsc);
822  }
823 
824  /* If Overrun error occurs, or if any error occurs in DMA mode reception,
825  consider error as blocking */
826  dmarequest = HAL_IS_BIT_SET(hsc->Instance->CR3, USART_CR3_DMAR);
827  if(((hsc->ErrorCode & HAL_SMARTCARD_ERROR_ORE) != RESET) || dmarequest)
828  {
829  /* Blocking error : transfer is aborted
830  Set the SMARTCARD state ready to be able to start again the process,
831  Disable Rx Interrupts, and disable Rx DMA request, if ongoing */
832  SMARTCARD_EndRxTransfer(hsc);
833  /* Disable the SMARTCARD DMA Rx request if enabled */
834  if(HAL_IS_BIT_SET(hsc->Instance->CR3, USART_CR3_DMAR))
835  {
836  CLEAR_BIT(hsc->Instance->CR3, USART_CR3_DMAR);
837 
838  /* Abort the SMARTCARD DMA Rx channel */
839  if(hsc->hdmarx != NULL)
840  {
841  /* Set the SMARTCARD DMA Abort callback :
842  will lead to call HAL_SMARTCARD_ErrorCallback() at end of DMA abort procedure */
843  hsc->hdmarx->XferAbortCallback = SMARTCARD_DMAAbortOnError;
844 
845  if(HAL_DMA_Abort_IT(hsc->hdmarx) != HAL_OK)
846  {
847  /* Call Directly XferAbortCallback function in case of error */
848  hsc->hdmarx->XferAbortCallback(hsc->hdmarx);
849  }
850  }
851  else
852  {
853  /* Call user error callback */
854  HAL_SMARTCARD_ErrorCallback(hsc);
855  }
856  }
857  else
858  {
859  /* Call user error callback */
860  HAL_SMARTCARD_ErrorCallback(hsc);
861  }
862  }
863  else
864  {
865  /* Call user error callback */
866  HAL_SMARTCARD_ErrorCallback(hsc);
867  hsc->ErrorCode = HAL_SMARTCARD_ERROR_NONE;
868  }
869  }
870  return;
871  } /* End if some error occurs */
872 
873  /* SMARTCARD in mode Transmitter -------------------------------------------*/
874  if(((isrflags & SMARTCARD_FLAG_TXE) != RESET) && ((cr1its & USART_CR1_TXEIE) != RESET))
875  {
876  SMARTCARD_Transmit_IT(hsc);
877  return;
878  }
879 
880  /* SMARTCARD in mode Transmitter (transmission end) ------------------------*/
881  if(((isrflags & SMARTCARD_FLAG_TC) != RESET) && ((cr1its & USART_CR1_TCIE) != RESET))
882  {
883  SMARTCARD_EndTransmit_IT(hsc);
884  return;
885  }
886 }
