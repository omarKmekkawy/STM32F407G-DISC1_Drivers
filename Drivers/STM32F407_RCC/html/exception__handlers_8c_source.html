92 // The values of BFAR and MMFAR stay unchanged if the BFARVALID or
93 // MMARVALID is set. However, if a new fault occurs during the
94 // execution of this fault handler, the value of the BFAR and MMFAR
95 // could potentially be erased. In order to ensure the fault addresses
96 // accessed are valid, the following procedure should be used:
97 // 1. Read BFAR/MMFAR.
98 // 2. Read CFSR to get BFARVALID or MMARVALID. If the value is 0, the
99 // value of BFAR or MMFAR accessed can be invalid and can be discarded.
100 // 3. Optionally clear BFARVALID or MMARVALID.
101 // (See Joseph Yiu's book).

174 int
175 isSemihosting (ExceptionStackFrame* frame, uint16_t opCode)
176 {
177  uint16_t* pw = (uint16_t*) frame->pc;
178  if (*pw == opCode)
179  {
180  uint32_t r0 = frame->r0;
181 #if defined(OS_DEBUG_SEMIHOSTING_FAULTS) || defined(OS_USE_SEMIHOSTING) || defined(OS_USE_TRACE_SEMIHOSTING_STDOUT)
182  uint32_t r1 = frame->r1;
183 #endif
184 #if defined(OS_USE_SEMIHOSTING) || defined(OS_USE_TRACE_SEMIHOSTING_STDOUT)
185  uint32_t* blk = (uint32_t*) r1;
186 #endif
187 
188 #if defined(OS_DEBUG_SEMIHOSTING_FAULTS)
189  // trace_printf ("sh r0=%d\n", r0);
190 #endif
191 
192  switch (r0)
193  {
194 
195 #if defined(OS_USE_SEMIHOSTING)
196 
197  case SEMIHOSTING_SYS_CLOCK:
198  case SEMIHOSTING_SYS_ELAPSED:
199  case SEMIHOSTING_SYS_FLEN:
200  case SEMIHOSTING_SYS_GET_CMDLINE:
201  case SEMIHOSTING_SYS_REMOVE:
202  case SEMIHOSTING_SYS_RENAME:
203  case SEMIHOSTING_SYS_SEEK:
204  case SEMIHOSTING_SYS_SYSTEM:
205  case SEMIHOSTING_SYS_TICKFREQ:
206  case SEMIHOSTING_SYS_TMPNAM:
207  case SEMIHOSTING_SYS_ISTTY:
208  frame->r0 = (uint32_t)-1; // the call is not successful or not supported
209  break;
210 
211  case SEMIHOSTING_SYS_CLOSE:
212  frame->r0 = 0; // call is successful
213  break;
214 
215  case SEMIHOSTING_SYS_ERRNO:
216  frame->r0 = 0; // the value of the C library errno variable.
217  break;
218 
219  case SEMIHOSTING_SYS_HEAPINFO:
220  blk[0] = 0; // heap_base
221  blk[1] = 0; // heap_limit
222  blk[2] = 0; // stack_base
223  blk[3] = 0; // stack_limit
224  break;
225 
226  case SEMIHOSTING_SYS_ISERROR:
227  frame->r0 = 0; // 0 if the status word is not an error indication
228  break;
229 
230  case SEMIHOSTING_SYS_READ:
231  // If R0 contains the same value as word 3, the call has
232  // failed and EOF is assumed.
233  frame->r0 = blk[2];
234  break;
235 
236  case SEMIHOSTING_SYS_READC:
237  frame->r0 = '\0'; // the byte read from the console.
238  break;
239 
240  case SEMIHOSTING_SYS_TIME:
241  frame->r0 = 0; // the number of seconds since 00:00 January 1, 1970.
242  break;
243 
244  case SEMIHOSTING_ReportException:
245 
246  NVIC_SystemReset ();
247  // Should not reach here
248  return 0;
249 
250 #endif // defined(OS_USE_SEMIHOSTING)
251 
252 #if defined(OS_USE_SEMIHOSTING) || defined(OS_USE_TRACE_SEMIHOSTING_STDOUT)
253 
254 #define HANDLER_STDIN (1)
255 #define HANDLER_STDOUT (2)
256 #define HANDLER_STDERR (3)
257 
258  case SEMIHOSTING_SYS_OPEN:
259  // Process only standard io/out/err and return 1/2/3
260  if (strcmp ((char*) blk[0], ":tt") == 0)
261  {
262  if ((blk[1] == 0))
263  {
264  frame->r0 = HANDLER_STDIN;
265  break;
266  }
267  else if (blk[1] == 4)
268  {
269  frame->r0 = HANDLER_STDOUT;
270  break;
271  }
272  else if (blk[1] == 8)
273  {
274  frame->r0 = HANDLER_STDERR;
275  break;
276  }
277  }
278  frame->r0 = (uint32_t)-1; // the call is not successful or not supported
279  break;
280 
281  case SEMIHOSTING_SYS_WRITE:
282  // Silently ignore writes to stdout/stderr, fail on all other handler.
283  if ((blk[0] == HANDLER_STDOUT) || (blk[0] == HANDLER_STDERR))
284  {
285 #if defined(OS_DEBUG_SEMIHOSTING_FAULTS)
286  frame->r0 = (uint32_t) blk[2]
287  - trace_write ((char*) blk[1], blk[2]);
288 #else
289  frame->r0 = 0; // all sent, no more.
290 #endif // defined(OS_DEBUG_SEMIHOSTING_FAULTS)
291  }
292  else
293  {
294  // If other handler, return the total number of bytes
295  // as the number of bytes that are not written.
296  frame->r0 = blk[2];
297  }
298  break;
299 
300 #endif // defined(OS_USE_SEMIHOSTING) || defined(OS_USE_TRACE_SEMIHOSTING_STDOUT)
301 
302 #if defined(OS_USE_SEMIHOSTING) || defined(OS_USE_TRACE_SEMIHOSTING_STDOUT) || defined(OS_USE_TRACE_SEMIHOSTING_DEBUG)
303 
304  case SEMIHOSTING_SYS_WRITEC:
305 #if defined(OS_DEBUG_SEMIHOSTING_FAULTS)
306  {
307  char ch = *((char*) r1);
308  trace_write (&ch, 1);
309  }
310 #endif
311  // Register R0 is corrupted.
312  break;
313 
314  case SEMIHOSTING_SYS_WRITE0:
315 #if defined(OS_DEBUG_SEMIHOSTING_FAULTS)
316  {
317  char* p = ((char*) r1);
318  trace_write (p, strlen (p));
319  }
320 #endif
321  // Register R0 is corrupted.
322  break;
323 
324 #endif
325 
326  default:
327  return 0;
328  }
329 
330  // Alter the PC to make the exception returns to
331  // the instruction after the faulty BKPT.
332  frame->pc += 2;
333  return 1;
334  }
335  return 0;
336 }
337 
338 #endif

340 // Hard Fault handler wrapper in assembly.
341 // It extracts the location of stack frame and passes it to handler
342 // in C as a pointer. We also pass the LR value as second
343 // parameter.
344 // (Based on Joseph Yiu's, The Definitive Guide to ARM Cortex-M3 and
345 // Cortex-M4 Processors, Third Edition, Chap. 12.8, page 402).
346 
347 void __attribute__ ((section(".after_vectors"),weak,naked))
348 HardFault_Handler (void)
349 {
350  asm volatile(
351  " tst lr,#4 \n"
352  " ite eq \n"
353  " mrseq r0,msp \n"
354  " mrsne r0,psp \n"
355  " mov r1,lr \n"
356  " ldr r2,=HardFault_Handler_C \n"
357  " bx r2"
358 
359  : /* Outputs */
360  : /* Inputs */
361  : /* Clobbers */
362  );
363 }

365 void __attribute__ ((section(".after_vectors"),weak,used))
366 HardFault_Handler_C (ExceptionStackFrame* frame __attribute__((unused)),
367  uint32_t lr __attribute__((unused)))
368 {
369 #if defined(TRACE)
370  uint32_t mmfar = SCB->MMFAR; // MemManage Fault Address
371  uint32_t bfar = SCB->BFAR; // Bus Fault Address
372  uint32_t cfsr = SCB->CFSR; // Configurable Fault Status Registers
373 #endif
374 
375 #if defined(OS_USE_SEMIHOSTING) || defined(OS_USE_TRACE_SEMIHOSTING_STDOUT) || defined(OS_USE_TRACE_SEMIHOSTING_DEBUG)
376 
377  // If the BKPT instruction is executed with C_DEBUGEN == 0 and MON_EN == 0,
378  // it will cause the processor to enter a HardFault exception, with DEBUGEVT
379  // in the Hard Fault Status register (HFSR) set to 1, and BKPT in the
380  // Debug Fault Status register (DFSR) also set to 1.
381 
382  if (((SCB->DFSR & SCB_DFSR_BKPT_Msk) != 0)
383  && ((SCB->HFSR & SCB_HFSR_DEBUGEVT_Msk) != 0))
384  {
385  if (isSemihosting (frame, 0xBE00 + (AngelSWI & 0xFF)))
386  {
387  // Clear the exception cause in exception status.
388  SCB->HFSR = SCB_HFSR_DEBUGEVT_Msk;
389 
390  // Continue after the BKPT
391  return;
392  }
393  }
394 
395 #endif
396 
397 #if defined(TRACE)
398  trace_printf ("[HardFault]\n");
399  dumpExceptionStack (frame, cfsr, mmfar, bfar, lr);
400 #endif // defined(TRACE)
401 
402 #if defined(DEBUG)
403  __DEBUG_BKPT();
404 #endif
405  while (1)
406  {
407  }
408 }
409 
410 #endif // defined(__ARM_ARCH_7M__) || defined(__ARM_ARCH_7EM__)
