353 void HAL_RNG_IRQHandler(RNG_HandleTypeDef *hrng)
354 {
355  /* RNG clock error interrupt occurred */
356  if((__HAL_RNG_GET_IT(hrng, RNG_IT_CEI) != RESET) || (__HAL_RNG_GET_IT(hrng, RNG_IT_SEI) != RESET))
357  {
358  /* Change RNG peripheral state */
359  hrng->State = HAL_RNG_STATE_ERROR;
360 
361  HAL_RNG_ErrorCallback(hrng);
362 
363  /* Clear the clock error flag */
364  __HAL_RNG_CLEAR_IT(hrng, RNG_IT_CEI|RNG_IT_SEI);
365 
366  }
367 
368  /* Check RNG data ready interrupt occurred */
369  if(__HAL_RNG_GET_IT(hrng, RNG_IT_DRDY) != RESET)
370  {
371  /* Generate random number once, so disable the IT */
372  __HAL_RNG_DISABLE_IT(hrng);
373 
374  /* Get the 32bit Random number (DRDY flag automatically cleared) */
375  hrng->RandomNumber = hrng->Instance->DR;
376 
377  if(hrng->State != HAL_RNG_STATE_ERROR)
378  {
379  /* Change RNG peripheral state */
380  hrng->State = HAL_RNG_STATE_READY;
381 
382  /* Data Ready callback */
