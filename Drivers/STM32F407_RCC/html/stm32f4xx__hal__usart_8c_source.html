1209  else
1210  {
1211  USART_TransmitReceive_IT(husart);
1212  }
1213  return;
1214  }
1215  }
1216  /* If some errors occur */
1217  if((errorflags != RESET) && ((cr3its & (USART_CR3_EIE | USART_CR1_PEIE)) != RESET))
1218  {
1219  /* USART parity error interrupt occurred ----------------------------------*/
1220  if(((isrflags & USART_SR_PE) != RESET) && ((cr1its & USART_CR1_PEIE) != RESET))
1221  {
1222  husart->ErrorCode |= HAL_USART_ERROR_PE;
1223  }
1224 
1225  /* USART noise error interrupt occurred --------------------------------*/

1227  {
1228  husart->ErrorCode |= HAL_USART_ERROR_NE;

1233  {
1234  husart->ErrorCode |= HAL_USART_ERROR_FE;

1239  {
1240  husart->ErrorCode |= HAL_USART_ERROR_ORE;
1241  }
1242

1244  {
1245  /* USART in mode Receiver -----------------------------------------------*/
1246  if(((isrflags & USART_SR_RXNE) != RESET) && ((cr1its & USART_CR1_RXNEIE) != RESET))
1247  {
1248  if(husart->State == HAL_USART_STATE_BUSY_RX)
1249  {
1250  USART_Receive_IT(husart);
1251  }
1252  else
1253  {
1254  USART_TransmitReceive_IT(husart);
1255  }
1256  }

1258  consider error as blocking */
1259  dmarequest = HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAR);

1265

1267  if (HAL_IS_BIT_SET(husart->Instance->CR3, USART_CR3_DMAR))
1268  {
1269  CLEAR_BIT(husart->Instance->CR3, USART_CR3_DMAR);
1270

1273  {

1275  will lead to call HAL_USART_ErrorCallback() at end of DMA abort procedure */
1276  husart->hdmarx->XferAbortCallback = USART_DMAAbortOnError;
1277

1281  husart->hdmarx->XferAbortCallback(husart->hdmarx);
1282  }
1283  }
1284  else
1285  {
1286  /* Call user error callback */
1287  HAL_USART_ErrorCallback(husart);
1288  }
1289  }
1290  else
1291  {
1292  /* Call user error callback */
1293  HAL_USART_ErrorCallback(husart);
1294  }
1295  }
1296  else
1297  {
1298  /* Call user error callback */
1299  HAL_USART_ErrorCallback(husart);
1300  husart->ErrorCode = HAL_USART_ERROR_NONE;
1301  }
1302  }
1303  return;
1304  }
1305 
1306  /* USART in mode Transmitter -----------------------------------------------*/
1307  if(((isrflags & USART_SR_TXE) != RESET) && ((cr1its & USART_CR1_TXEIE) != RESET))
1308  {
1309  if(husart->State == HAL_USART_STATE_BUSY_TX)
1310  {

1312  }
1313  else
