490 void HAL_DCMI_IRQHandler(DCMI_HandleTypeDef *hdcmi)
491 {
492  uint32_t isr_value = READ_REG(hdcmi->Instance->MISR);
493 
494  /* Synchronization error interrupt management *******************************/
495  if((isr_value & DCMI_FLAG_ERRRI) == DCMI_FLAG_ERRRI)
496  {
497  /* Clear the Synchronization error flag */
498  __HAL_DCMI_CLEAR_FLAG(hdcmi, DCMI_FLAG_ERRRI);
499 
500  /* Update error code */
501  hdcmi->ErrorCode |= HAL_DCMI_ERROR_SYNC;
502 
503  /* Change DCMI state */
504  hdcmi->State = HAL_DCMI_STATE_ERROR;
505 
506  /* Set the synchronization error callback */
507  hdcmi->DMA_Handle->XferAbortCallback = DCMI_DMAError;
508 
509  /* Abort the DMA Transfer */
510  HAL_DMA_Abort_IT(hdcmi->DMA_Handle);
511  }
512  /* Overflow interrupt management ********************************************/
513  if((isr_value & DCMI_FLAG_OVRRI) == DCMI_FLAG_OVRRI)
514  {
515  /* Clear the Overflow flag */
516  __HAL_DCMI_CLEAR_FLAG(hdcmi, DCMI_FLAG_OVRRI);
517 
518  /* Update error code */
519  hdcmi->ErrorCode |= HAL_DCMI_ERROR_OVR;
520 
521  /* Change DCMI state */
522  hdcmi->State = HAL_DCMI_STATE_ERROR;
523 
524  /* Set the overflow callback */
525  hdcmi->DMA_Handle->XferAbortCallback = DCMI_DMAError;
526 
527  /* Abort the DMA Transfer */
528  HAL_DMA_Abort_IT(hdcmi->DMA_Handle);
529  }
530  /* Line Interrupt management ************************************************/
531  if((isr_value & DCMI_FLAG_LINERI) == DCMI_FLAG_LINERI)
532  {
533  /* Clear the Line interrupt flag */
534  __HAL_DCMI_CLEAR_FLAG(hdcmi, DCMI_FLAG_LINERI);
535 
536  /* Line interrupt Callback */
537  HAL_DCMI_LineEventCallback(hdcmi);
538  }
539  /* VSYNC interrupt management ***********************************************/
540  if((isr_value & DCMI_FLAG_VSYNCRI) == DCMI_FLAG_VSYNCRI)
541  {
542  /* Clear the VSYNC flag */
543  __HAL_DCMI_CLEAR_FLAG(hdcmi, DCMI_FLAG_VSYNCRI);
544 
545  /* VSYNC Callback */
546  HAL_DCMI_VsyncEventCallback(hdcmi);
547  }
548  /* FRAME interrupt management ***********************************************/
549  if((isr_value & DCMI_FLAG_FRAMERI) == DCMI_FLAG_FRAMERI)
550  {
551  /* When snapshot mode, disable Vsync, Error and Overrun interrupts */
552  if((hdcmi->Instance->CR & DCMI_CR_CM) == DCMI_MODE_SNAPSHOT)
553  {
554  /* Disable the Line, Vsync, Error and Overrun interrupts */
555  __HAL_DCMI_DISABLE_IT(hdcmi, DCMI_IT_LINE | DCMI_IT_VSYNC | DCMI_IT_ERR | DCMI_IT_OVR);
556  }
557 
558  /* Disable the Frame interrupt */
559  __HAL_DCMI_DISABLE_IT(hdcmi, DCMI_IT_FRAME);
560 
561  /* Frame Callback */
562  HAL_DCMI_FrameEventCallback(hdcmi);
563  }
564 }

804  HAL_DMAEx_ChangeMemory(hdcmi->DMA_Handle, (tmp + (8U*hdcmi->XferSize)), MEMORY0);

806  }
807  /* Update memory 1 address location */

809  {

811  HAL_DMAEx_ChangeMemory(hdcmi->DMA_Handle, (tmp + (8U*hdcmi->XferSize)), MEMORY1);
812  hdcmi->XferCount--;
813  }
814  }
815  /* Update memory 0 address location */
816  else if((hdcmi->DMA_Handle->Instance->CR & DMA_SxCR_CT) != 0U)
817  {
818  hdcmi->DMA_Handle->Instance->M0AR = hdcmi->pBuffPtr;
819  }
820  /* Update memory 1 address location */
821  else if((hdcmi->DMA_Handle->Instance->CR & DMA_SxCR_CT) == 0U)
822  {
823  tmp = hdcmi->pBuffPtr;
824  hdcmi->DMA_Handle->Instance->M1AR = (tmp + (4U*hdcmi->XferSize));
